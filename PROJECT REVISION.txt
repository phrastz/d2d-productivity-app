function calculateProjectProgress(
  subProjects: SubProject[],
  directTasks: Task[]
): number {
  // Hitung dari Sub-Projects (dengan bobot)
  const subProjectProgress = subProjects.reduce((sum, sp) => {
    return sum + (sp.progress_percent * sp.weight_contribution);
  }, 0);
  
  const totalSubWeight = subProjects.reduce((sum, sp) => 
    sum + sp.weight_contribution, 0
  );
  
  // Hitung dari Direct Tasks (tanpa sub-project)
  const directTaskProgress = directTasks.reduce((sum, task) => {
    return sum + task.progress_percent;
  }, 0);
  
  const totalItems = subProjects.length + directTasks.length;
  if (totalItems === 0) return 0;
  
  // Average progress dari semua items
  const weightedSubProgress = totalSubWeight > 0 
    ? subProjectProgress / totalSubWeight * (subProjects.length / totalItems)
    : 0;
    
  const directProgress = directTasks.length > 0
    ? directTaskProgress / directTasks.length * (directTasks.length / totalItems)
    : 0;
  
  return parseFloat(((weightedSubProgress + directProgress) * 100).toFixed(2));
}

// API: DELETE /api/tasks/[taskId]
export async function DELETE(
  request: Request,
  { params }: { params: Promise<{ taskId: string }> }
) {
  const { taskId } = await params;
  
  const task = await db.task.findUnique({
    where: { id: taskId },
    select: { projectId: true, subProjectId: true }
  });
  
  if (!task) {
    return NextResponse.json({ error: 'Task not found' }, { status: 404 });
  }
  
  // Delete task + comments
  await db.task.delete({
    where: { id: taskId }
  });
  
  // Recalculate progress parent
  if (task.subProjectId) {
    // Task nested в†’ update sub-project progress
    await recalculateSubProjectProgress(task.subProjectId);
    await recalculateProjectProgress(task.projectId);
  } else {
    // Task standalone в†’ update project progress langsung
    await recalculateProjectProgress(task.projectId);
  }
  
  return NextResponse.json({ success: true });
}

Perhitungan Progress Berbasis Timeline
A. Time-Based Progress (Otomatis)
Progress bisa dihitung otomatis berdasarkan waktu yang sudah berlalu:
function calculateTimeBasedProgress(
  startDate: Date,
  endDate: Date,
  currentDate: Date = new Date()
): number {
  const totalDuration = endDate.getTime() - startDate.getTime();
  const elapsed = currentDate.getTime() - startDate.getTime();
  
  // Belum mulai
  if (currentDate < startDate) return 0;
  
  // Sudah lewat deadline
  if (currentDate > endDate) return 100;
  
  // Dalam progress
  const progress = (elapsed / totalDuration) * 100;
  return Math.min(100, Math.max(0, progress));
}

// Contoh usage:
const progress = calculateTimeBasedProgress(
  new Date("2025-01-01"),
  new Date("2025-01-31"),
  new Date("2025-01-15")  // Hari ini

function getProgressStatus(
  startDate: Date,
  endDate: Date,
  actualProgress: number
): {
  expectedProgress: number;
  status: "ON_TRACK" | "AHEAD" | "DELAYED";
  variance: number;
} {
  const expectedProgress = calculateTimeBasedProgress(startDate, endDate);
  const variance = actualProgress - expectedProgress;
  
  let status: "ON_TRACK" | "AHEAD" | "DELAYED";
  
  if (variance >= -10 && variance <= 10) {
    status = "ON_TRACK";      // В±10% tolerance
  } else if (variance > 10) {
    status = "AHEAD";         // Lebih cepat
  } else {
    status = "DELAYED";       // Terlambat
  }
  
  return {
    expectedProgress,
    status,
    variance
  };
}

// Contoh:
const status = getProgressStatus(
  new Date("2025-01-01"),
  new Date("2025-01-31"),
  30  // User input: baru 30% selesai
);
// Hasil: { expectedProgress: 50, status: "DELAYED", variance: -20 }
// в†’ Seharusnya 50%, tapi baru 30% в†’ DELAYED 20%

Aggregate Progress Calculation
Hierarchical Progress Roll-up:
async function recalculateProjectProgress(projectId: string) {
  // 1. Ambil semua sub-projects
  const subProjects = await db.subProject.findMany({
    where: { projectId },
    include: {
      tasks: true  // Nested tasks
    }
  });
  
  // 2. Ambil standalone tasks (langsung di project)
  const directTasks = await db.task.findMany({
    where: {
      projectId,
      subProjectId: null
    }
  });
  
  // 3. Hitung progress setiap sub-project dari nested tasks
  const subProjectsWithProgress = await Promise.all(
    subProjects.map(async (sp) => {
      const taskProgress = sp.tasks.reduce((sum, task) => 
        sum + task.progressPercent, 0
      );
      
      const avgProgress = sp.tasks.length > 0
        ? taskProgress / sp.tasks.length
        : 0;
      
      // Update sub-project progress
      await db.subProject.update({
        where: { id: sp.id },
        data: { progressPercent: Math.round(avgProgress) }
      });
      
      return {
        ...sp,
        calculatedProgress: avgProgress,
        weight: Number(sp.weightContribution)
      };
    })
  );
  
  // 4. Hitung progress direct tasks
  const directTaskProgress = directTasks.reduce((sum, task) => 
    sum + task.progressPercent, 0
  );
  
  const avgDirectTaskProgress = directTasks.length > 0
    ? directTaskProgress / directTasks.length
    : 0;
  
  // 5. Aggregate semua progress
  const totalWeight = subProjectsWithProgress.reduce((sum, sp) => 
    sum + sp.weight, 0
  );
  
  const weightedSubProgress = subProjectsWithProgress.reduce((sum, sp) => 
    sum + (sp.calculatedProgress * sp.weight), 0
  );
  
  // Hitung final project progress
  const totalItems = subProjects.length + directTasks.length;
  
  let finalProgress: number;
  
  if (totalItems === 0) {
    finalProgress = 0;
  } else {
    // Weighted average dari sub-projects + direct tasks
    const subProjectContribution = totalWeight > 0
      ? (weightedSubProgress / totalWeight) * (subProjects.length / totalItems)
      : 0;
      
    const directTaskContribution = directTasks.length > 0
      ? avgDirectTaskProgress * (directTasks.length / totalItems)
      : 0;
      
    finalProgress = subProjectContribution + directTaskContribution;
  }
  
  // 6. Update project
  await db.project.update({
    where: { id: projectId },
    data: {
      progressRatio: parseFloat((finalProgress / 100).toFixed(2)),
      updatedAt: new Date()
    }
  });
  
  return finalProgress;
}

interface TimelineItem {

  type: "PROJECT" | "SUB_PROJECT" | "TASK";
  startDate: Date;
  endDate: Date;
  progressPercent: number;
  status: "ON_TRACK" | "DELAYED" | "AHEAD";

// Fetch semua items untuk Gantt chart
async function getProjectTimeline(projectId: string) {
  const project = await db.project.findUnique({
    where: { id: projectId },
    include: {
      subProjects: {
        include: {
          tasks: true
        }
      },
      tasks: {
        where: { subProjectId: null }  // Hanya standalone tasks
      }
    }
  });
  
  const timeline: TimelineItem[] = [
    // Project level
    {
      id: project.id,
      name: project.name,
      type: "PROJECT",
      startDate: project.startDate,
      endDate: project.endDate,
      progressPercent: Number(project.progressRatio) * 100,
      status: getProgressStatus(
        project.startDate,
        project.endDate,
        Number(project.progressRatio) * 100
      ).status
    },
    
    // Sub-projects
    ...project.subProjects.map(sp => ({
      id: sp.id,
      name: sp.name,
      type: "SUB_PROJECT" as const,
      startDate: sp.startDate,
      endDate: sp.endDate,
      progressPercent: sp.progressPercent,
      status: getProgressStatus(
        sp.startDate,
        sp.endDate,
        sp.progressPercent
      ).status,
      parentId: project.id
    })),
    
    // Nested tasks (dalam sub-project)
    ...project.subProjects.flatMap(sp =>
      sp.tasks.map(task => ({
        id: task.id,
        name: task.name,
        type: "TASK" as const,
        startDate: task.startDate!,
        endDate: task.endDate!,
        progressPercent: task.progressPercent,
        status: getProgressStatus(
          task.startDate!,
          task.endDate!,
          task.progressPercent
        ).status,
        parentId: sp.id
      }))
    ),
    
    // Standalone tasks (langsung di project)
    ...project.tasks.map(task => ({
      id: task.id,
      name: task.name,
      type: "TASK" as const,
      startDate: task.startDate!,
      endDate: task.endDate!,
      progressPercent: task.progressPercent,
      status: getProgressStatus(
        task.startDate!,
        task.endDate!,
        task.progressPercent
      ).status,
      parentId: project.id
    }))
  ];
  
  return timeline;
}
